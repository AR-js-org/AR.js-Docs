We will start with a very basic example of the three.js API for location-based AR.js. This is assuming you install AR.js via npm and use a bundler such as Webpack to build. Here is a sample package.json:

{
    "dependencies": {
        "@ar-js-org/ar.js": "3.4.2",
    },
    "devDependencies": {
        "webpack": "^5.75.0",
        "webpack-cli": "^5.0.0"
    },
    "scripts": {
        "build": "npx webpack"
    }
}

and a sample webpack.config.js:

const path = require('path');

module.exports = {
    mode: 'development',
    entry: './index.js',
    output: {
        path: path.resolve(__dirname, 'dist'),
        filename: 'bundle.js'
    },
    optimization: {
        minimize: false
    }
};

This will build a bundle named bundle.js in the dist subdirectory from a source file index.js. Here is a simple index.js using the three.js location-based API:

import * as THREE from 'three';
import * as THREEx from './node_modules/@ar-js-org/ar.js/three.js/build/ar-threex-location-only.js'

function main() {

    const canvas = document.getElementById("canvas1");

    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(80, 1.33, 0.001, 10000);
    const renderer = new THREE.WebGLRenderer({
        canvas: canvas
    });

    const arjs = new THREEx.LocationBased(scene, camera);

    arjs.fakeGps(-0.72, 51.05);

    const webcam = new THREEx.WebcamRenderer(renderer, '#video1');

    const orientationControls = new THREEx.DeviceOrientationControls(camera);

    const red = new THREE.MeshBasicMaterial({color: 0xff0000});
    const box = new THREE.BoxGeometry();

    const mesh = new THREE.Mesh(box, red);

    arjs.add(mesh, -0.72, 51.051);

    function render() {
        const newWidth = canvas.clientWidth;
        const newHeight = canvas.clientHeight;

        if(newWidth != canvas.width || newHeight != canvas.height) {
            renderer.setSize(newWidth, newHeight, false);
            camera.aspect  = newWidth / newHeight;
            camera.updateProjectionMatrix();
        }
        orientationControls.update();
        webcam.update();
        renderer.render(scene, camera);
        requestAnimationFrame(render);

    }

    requestAnimationFrame(render);

    window.addEventListener("unload", e=> {
        console.log("Unload");
        webcam.dispose();
    });
}

main();

Much of this will be familiar if you have done basic three.js coding before. Focusing on the AR.js specific code:

const arjs = new THREEx.LocationBased(scene, camera);

This creates a new LocationBased manager object using the three.js scene and camera.

arjs.fakeGps(-0.72, 51.05);

This sends a fake GPS position, which is used here because we will add a mesh at a location very close to this fake GPS position. You can also start listening for real GPS positions using the startGps() method, e.g.:

arjs.startGps();

We then initialise the webcam:

const webcam = new THREEx.WebcamRenderer(renderer, '#video1');

The WebcamRenderer requires the three.js renderer object and the ID of an HTML5 video element as arguments. Next we initialise the device orientation controls:

const orientationControls = new THREEx.DeviceOrientationControls(camera);

We then create a mesh and add it to the LocationBased object at a given longitude and latitude:

arjs.add(mesh, -0.72, 51.051);

We need to ensure the webcam and device orientation controls update for each rendering frame. So we add this to our render() function:

orientationControls.update();
webcam.update();
